
#Send Teams to user

curl -X POST http://localhost:8080/chats/send \
  -H "Content-Type: application/json" \
  -d '{
    "sender_user_key": "<user_key from /auth/callback>",
    "to_upn": "someone@yourtenant.com",
    "message": "<b>Hello from delegated!</b>"
  }'


cat > chat.json <<'EOF'
{
  "recipient_upn": "ramassey@GPSEXperimentationSandbox.onmicrosoft.com",
  "message": "Hello via delegated chat API!"
}
EOF

curl -X POST http://localhost:8080/send-chat-delegated \
  -H "Content-Type: application/json" \
  --data-binary @chat.json
